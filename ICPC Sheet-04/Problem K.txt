#include <stdio.h>
#include <string.h>
int main() {
    int N;
    scanf("%d", &N);

    for (int t = 0; t < N; t++) {
        char S[55], T[55];
        scanf("%s %s", S, T);
        int lenS = strlen(S);
        int lenT = strlen(T);

        int maxLen;
        if (lenS > lenT) {
            maxLen = lenS;
        } else {
            maxLen = lenT;
        }
        for (int i = 0; i < maxLen; i++) {
            if (i < lenS){
                    printf("%c", S[i]);
            }

            if (i < lenT){
                    printf("%c", T[i]);
            }
        }
        printf("\n");
    }
}
