#include <stdio.h>
#include <string.h>
int main() {
    char S[1005];
    scanf("%s", S);
    int len = strlen(S);
    for (int i = 0; i < len; ) {
        if (strncmp(&S[i], "EGYPT", 5) == 0) {
            printf(" ");  
            i += 5; 
        } else {
            printf("%c", S[i]);
            i++;
        }
    }

    printf("\n");
}
