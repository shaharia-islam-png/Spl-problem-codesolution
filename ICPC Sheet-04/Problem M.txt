#include <stdio.h>
#include <string.h>
int main() {
    char s[10005];
    scanf("%s", s);
    char target[] = "hello";
    int j = 0;
    for(int i = 0; s[i]; i++){
        if(s[i] == target[j])
            j++;
        if(j == 5){
             break;
        }
    }
    if(j == 5){
        printf("YES\n");
    }
    else{
         printf("NO\n");   
    }
}
