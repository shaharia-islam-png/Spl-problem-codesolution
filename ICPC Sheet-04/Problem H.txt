#include <stdio.h>
#include <string.h>
int main(){
    int T;
    scanf("%d", &T);
    for(int k = 0;k< T; k++){
        char s[100005];
        scanf("%s", s);

        int len = strlen(s);
        int good = 0;
        for(int i = 0; i < len - 2; i++) {
            if(s[i] == s[i+2] && s[i] != s[i+1]) {
                good = 1;
                break;
            }
        }
        if(good){
           printf("Good\n");   
        }
        else{
            printf("Bad\n");
        }
            
    }
}
