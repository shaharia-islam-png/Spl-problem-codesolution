#include <stdio.h>
int main() {
    char S[100005];
    scanf("%s", S);
    for(int i = 0; S[i]; i++){
        char c = S[i];
        if(c == ','){
            printf(" ");
        }
        else if(c >= 'a' && c <= 'z'){
            printf("%c", c - 'a' + 'A');
        }
        else if(c >= 'A' && c <= 'Z'){
            printf("%c", c - 'A' + 'a');
        }
        else{
            printf("%c", c);
        }
    }
    printf("\n");
}
