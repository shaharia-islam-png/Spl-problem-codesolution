#include <stdio.h>
int main() {
    char s[1000005];
   scanf("%[^\n]", s);
    int count = 0;
    int word = 0;
    for (int i = 0; s[i]; i++) {
        if ((s[i] >= 'a' && s[i] <= 'z') ||
            (s[i] >= 'A' && s[i] <= 'Z')) {

            if (word == 0) {
                count++;
                word = 1;
            }
        }
        else{
            word = 0;
        }
    }
    printf("%d\n", count);
}
