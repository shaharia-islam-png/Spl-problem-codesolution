#include <stdio.h>
#include <math.h>
int main() {
    long long N;
    scanf("%lld", &N);

    int isPrime = 1;
    if (N <= 1) isPrime = 0;
    for (long long i = 2; i <= sqrt(N); i++){
        if (N % i == 0) {
            isPrime = 0;
            break;
        }
    }
    if (isPrime){
      printf("YES\n");
    }
    else{
      printf("NO\n");
    }
}
