#include <stdio.h>
void average() {
        int n;
    scanf("%d", &n);
    double a[n];
    for(int i = 0; i < n; i++) {
        scanf("%lf", &a[i]);
    }
    double sum = 0.0;
    for(int i = 0; i < n; i++) {
        sum = sum + a[i];
    }
    double avg = sum / n;
    printf("%.7lf", avg);
}

int main() {

    average();

}
