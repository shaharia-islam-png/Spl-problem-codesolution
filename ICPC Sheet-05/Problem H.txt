#include <stdio.h>
int printCharN(int N, char C) {
    for (int i = 0; i < N; i++) {
        printf("%c", C);
        if (i != N - 1) printf(" ");
    }
    printf("\n");
    return 0;
}

int main() {
    int T;
    scanf("%d", &T);
    for (int t = 0; t < T; t++) {
        int N;
        char C;
        scanf("%d %c", &N, &C);
        printCharN(N, C);
    }
}
