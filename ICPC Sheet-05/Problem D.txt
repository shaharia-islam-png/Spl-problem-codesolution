#include <stdio.h>
#include <math.h>
void prime(long long N){
    int isPrime = 1;
    if (N <= 1) isPrime = 0;
    for (long long i = 2; i <= sqrt(N); i++){
        if (N % i == 0) {
            isPrime = 0;
            break;
        }
    }

    if (isPrime)
        printf("YES\n");
    else
        printf("NO\n");
}

int main(){
    int T;
    scanf("%d", &T);

    for (int t = 0; t < T; t++){
        long long N;
        scanf("%lld", &N);
        prime(N);
    }
}
