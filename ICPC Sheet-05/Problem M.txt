#include <stdio.h>
int Distinct(int arr[], int n) {
    int count = 0;
    for (int i = 0; i < n; i++) {
        int isDistinct = 1;
        for (int j = 0; j < i; j++) {
            if (arr[i] == arr[j]) {
                isDistinct = 0;
                break;
            }
        }
        if (isDistinct)
            count++;
    }

    return count;
}

int main() {
    int N;
    scanf("%d", &N);

    int A[N];
    for (int i = 0; i < N; i++) {
        scanf("%d", &A[i]);
    }
    int result = Distinct(A, N);
    printf("%d\n", result);

}
