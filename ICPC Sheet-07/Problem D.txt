#include <stdio.h>

int loop(long long n) {
    if (n == 0)
        return 0;
    loop(n / 10);
    printf("%lld ", n % 10);
}

int main() {
    int T;
    scanf("%d", &T);
    for (int i = 1; i <= T; i++) {
        long long N;
        scanf("%lld", &N);

        if (N == 0)
            printf("0 ");
        else
            loop(N);

        printf("\n");
    }

    return 0;
}
