#include <stdio.h>

int pyramid(int N, int row) {
    if (row == N)   // base case
        return 0;
    for (int i = 0; i < N - row - 1; i++)
        printf(" ");
    for (int i = 0; i < 2 * row + 1; i++)
        printf("*");
    printf("\n");
    pyramid(N, row + 1);
}
int main() {
    int N;
    scanf("%d", &N);

    pyramid(N, 0);
}
