#include <stdio.h>

long long int sumarray(long long arr[], int n) {
    if (n == 0)
        return 0;
    return arr[n - 1] + sumarray(arr, n - 1);
}

int main() {
    int N;
    scanf("%d", &N);
    long long A[N];
    for (int i = 0; i < N; i++){
        scanf("%lld", &A[i]);
    }
    printf("%lld\n", sumarray(A, N));
}
