#include <stdio.h>

int R, C;
int A[105][105], B[105][105];

void addMatrix(int i, int j) {
    if (i == R)
        return;

    if (j == C){ 
        printf("\n");
        addMatrix(i + 1, 0);
        return;
    }

    printf("%d ", A[i][j] + B[i][j]);
    addMatrix(i, j + 1);
}

int main() {
    scanf("%d %d", &R, &C);

    for (int i = 0; i < R; i++)
        for (int j = 0; j < C; j++)
            scanf("%d", &A[i][j]);

    for (int i = 0; i < R; i++)
        for (int j = 0; j < C; j++)
            scanf("%d", &B[i][j]);

    addMatrix(0, 0);
}
