#include <stdio.h>
int binary(long long n) {
    if (n == 0)
        return 0;
    binary(n / 2);
    printf("%lld", n % 2);
}
int main() {
    int T;
    scanf("%d", &T);

    for (int i = 1; i <= T; i++) {
        long long N;
        scanf("%lld", &N);
        if (N == 0)
            printf("0");
        else
            binary(N);

        printf("\n");
    }
}
