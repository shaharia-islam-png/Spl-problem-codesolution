#include <stdio.h>
int printpyramid(int N, int row) {
    if (row == N)
        return 0;
    for (int s = 0; s < row; s++)
        printf(" ");
    int stars = 2 * (N - row) - 1;
    for (int j = 0; j < stars; j++)
        printf("*");

    printf("\n");
    printpyramid(N, row + 1);
}

int main() {
    int N;
    scanf("%d", &N);

    printpyramid(N, 0);

    return 0;
}
