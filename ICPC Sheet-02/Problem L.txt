#include <stdio.h>
int main() {
    int a, b, gcd;
    scanf("%d %d",&a,&b);
    int min ;
    if(a < b){
        min=a;
    }
    else{
        min=b;
    }
    for(int i = min; i >= 1; i--) {
        if(a %i == 0 && b%i == 0){
            gcd = i;
            break;
        }
    }

    printf("%d\n",gcd);
}
