#include <stdio.h>
int main() {
    int k,s;
    scanf("%d %d", &k,&s);
    int count =0;
    for (int j = 0;  j<=k; j++){
        for (int i = 0; i <= k;i++) {
            int z = s-(i +j);
            if (z >= 0 && z <= k){
                count++;
            }
    }
    }
    printf("%d\n", count);
}
