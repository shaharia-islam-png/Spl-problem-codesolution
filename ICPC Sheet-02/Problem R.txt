#include <stdio.h>
int main() {
    int N, M;
    for(int i=1;;i++){
        scanf("%d %d", &N, &M);

        if(N <= 0 || M <= 0)
            break;

        int sum = 0;

        if(N <= M) {
            for(int i = N; i <= M; i++) {
                printf("%d ", i);
                sum =sum+i;
            }
        }
        else {
            for(int i = M; i <= N; i++) {
                printf("%d ", i);
                 sum =sum+i;
            }
        }

        printf("sum =%d\n", sum);
    }
}
