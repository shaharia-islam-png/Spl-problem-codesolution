#include <stdio.h>
int main() {
    int n;
    scanf("%d", &n);
    int a[n];
    for(int i = 0; i < n; i++) {
        scanf("%d", &a[i]);
    }

    int min = a[0], max = a[0];
    int minindex = 0, maxindex = 0;

    for(int i = 0; i < n; i++) {
        if(a[i] < min) {
            min = a[i];
            minindex = i;
        }
        if(a[i] > max) {
            max = a[i];
            maxindex = i;
        }
    }
    int x = a[minindex];
    a[minindex] = a[maxindex];
    a[maxindex] = x;
    for(int i = 0; i < n; i++) {
        printf("%d ", a[i]);
    }
}
