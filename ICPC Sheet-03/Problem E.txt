#include <stdio.h>
int main() {
    int N;
    scanf("%d", &N);
    int A[N];
    for (int i = 0; i < N; i++) {
        scanf("%d", &A[i]);
   }
    int minvalue = A[0];
    int minpos = 1;
    for (int i = 1; i < N; i++){
        if (A[i] < minvalue) {
            minvalue = A[i];
            minpos = i + 1;
        }
    }
    printf("%d %d\n", minvalue, minpos);

    return 0;
}
