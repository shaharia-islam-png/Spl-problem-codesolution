#include <stdio.h>
int main() {
    int n,m;
    scanf("%d %d", &n,&m);
    int A[n][m];
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < m; j++) {
            scanf("%d", &A[i][j]);
        }
    }
    int X;
    scanf("%d", &X);

    int found = 0;
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < m; j++) {
            if (A[i][j] == X) {
                found = 1;
                break;
            }
        }
    }
    if (found == 1){
        printf("will not take number");
    }
    else{
        printf("will take number");
    }

}
