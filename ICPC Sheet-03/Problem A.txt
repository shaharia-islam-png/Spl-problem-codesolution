#include <stdio.h>
int main(){
    long long n;
    scanf("%lld", &n);
    long long a[n];
    long long sum =0;
    for(long long i = 0; i < n; i++){
        scanf("%lld", &a[i]);
    }

    for(long long i = 0; i < n; i++) {
        sum = sum+a[i];
    }
     if(sum < 0){
            sum = -sum;
     }
    printf("%lld\n", sum);

    return 0;
}
