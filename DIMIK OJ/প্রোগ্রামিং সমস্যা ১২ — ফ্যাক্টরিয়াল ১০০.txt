#include <stdio.h>
int main() {
    int t;
    scanf("%d", &t);

    for(int i = 0; i < t; i++){
        int n;
        scanf("%d", &n);
        int zeros = 0;
        for(int f = 5; f <= n; f =f* 5){
            zeros = zeros + ( n / f);
        }

        printf("%d\n", zeros);
    }
}
