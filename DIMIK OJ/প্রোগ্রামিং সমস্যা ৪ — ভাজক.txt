#include<stdio.h>
int main() {
    int t, n;
    scanf("%d", &t);
    for (int c = 1; c <= t; c++) {
        scanf("%d", &n);
        printf("Case %d: ", c);

        for (int i = 1; i <= n; i++) {
            if (n % i == 0) {
                printf("%d", i);
                if (i != n) printf(" ");
            }
        }
        printf("\n");
    }
}
