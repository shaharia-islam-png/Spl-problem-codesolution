#include <stdio.h>
int main() {
    int num,ev= 0, odd = 0, pos= 0, neg= 0;
    for(int i = 1;i<= 5;i++){
    scanf("%d", &num);
    if(num % 2== 0){
          ev++;
    }
     else {
         odd++;
        }
    if(num > 0) {
          pos++;
        }
    else if(num < 0) {
          neg++;
        }
    }
    printf("%d valor(es) par(es)\n", ev);
    printf("%d valor(es) impar(es)\n", odd);
    printf("%d valor(es) positivo(s)\n", pos);
    printf("%d valor(es) negativo(s)\n", neg);
}
