#include <stdio.h>
int main() {
    int number, hours;
    double rate;
    scanf("%d", &number);
    scanf("%d", &hours);
    scanf("%lf", &rate);
    
    double salary = hours * rate;

    printf("NUMBER = %d\n", number);
    printf("SALARY = U$ %.2lf\n", salary);

    return 0;
}
