#include <stdio.h>
int main() {
    int code1, units1;
    double price1;
    int code2, units2;
    double price2;
    scanf("%d %d %lf", &code1, &units1, &price1);
    scanf("%d %d %lf", &code2, &units2, &price2);
    double total = units1*price1 + units2*price2;
    printf("VALOR A PAGAR: R$ %.2lf\n", total);

    return 0;
}
